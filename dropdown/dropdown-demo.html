ToFishes系列jQuery插件 - Dropdown下拉列表

点击下载 DEMO+源码

日期： 2014年09月09日

@require 该插件实现基于jQuery v1.11.1

使用示例：

下拉选项：

  • 苹果
  • 橘子
  • 香蕉
<!-- 示例的HTML结构 -->
<div class="dropdown">
    <p class="dropdown-show-wrap">
        下拉选项：<span class="dropdown-show"></span>
    </p>
    <ul class="dropdown-list">
        <!-- class=active的项目将作为默认选中项，否则默认选中第一个 -->
        <li data-value="1">苹果</li>
        <li data-value="2" class="active">橘子</li>
        <li data-value="3">香蕉</li>
    </ul>
</div>

/* 示例中所需要的css */
.dropdown {
    position: relative;
    width: 200px;
}
.dropdown-show-wrap {
    padding: 5px 10px;
    border:1px solid #ccc;
    box-shadow: 0 0 3px #ccc inset;
}
.dropdown-show {
    position: relative;
    display: inline-block;
    min-width: 100px;
}
.dropdown-show-wrap:after {
    position: absolute;
    right: 5px;
    top: 5px;
    content: '\25BC';
    color: #888;
}
.active .dropdown-show-wrap:after {
    content: '\25B2';
}
.dropdown-list {
    display: none;
    position: absolute;
    width: 198px;
    background: #fff;
    border: 1px solid #ccc;
}
.dropdown-list li {
    padding:5px 10px;
    cursor: pointer;
}
.dropdown-list li:hover {
    background: #e8e8e8;
}
.dropdown-list li.active {
    background: #09d;
    color: #fff;
}

// 调用代码
// 以下展示的参数为默认参数，可选
$('.dropdown').dropdown({
    'show': '.dropdown-show', // 显示文字的容器
    'list': '.dropdown-list', // 下拉列表容器
    'item': 'li', // 下拉列表单项项目
    'onselect': function ($selected, $show) {
        window.console & console.info($selected.data('value'));
    } // 选中后的回调，可以返回字符串作为显示文字
});
