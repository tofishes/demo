调用代码：
	var preview = $("#preview");
    //自定义需要，按参数默认值规范好的单选、多选及下拉，这样调用代码会少很多
    $(function(){   
        $(".jradio").jRadio({
            input:".jvalue",
            onchange:function(value){
                preview.html("单选值被改变，当前选中值： " + value);
            }
        });
        $(".jcheckbox").jCheckbox({
            onchange:function(value){
                preview.html("多选值被改变，当前选中值为以逗号相隔的字符串： \"" + value + "\"");
            }		
        });
        $(".jselect").jSelect({
            con:".db-con",
            attr: "name", //这里不使用默认值，因为li的value属性只能被获取到数值类型
            width: -1, //同显示区同宽
            widthFix:6, //这里修正6像素宽度
            onchange: function(value,index){
                preview.html("下拉第" + index + "项被选中，当前选中值： " + value);
            }
        });
    });
